// function prototypes
int Sum(int);

// program entry
int main (void)
{
	int n = 0;
	int i = 0;
		
	printf("\nPlease enter a positive integer n: ");
	scanf("%d", &n);
	
	int result = Sum(n);

	printf("\nSum(%d) = ", n);
	
	for(i = n; i > 1; i--)
	{
		printf("%d+", i);
	}
	printf("1 = %d\n\n", result);

	return 0;
}




/******************************************************************************
FUNCTION		: Sum

DESCRIPTION		: This function returns the sum of integers from 1 to n:
				  Ex: 1 + ... + (n - 2) + (n -1) + n

INPUT			: Type			: int
				: Description	: n
				
OUTPUT			: Type 			: int
				: Description	: the sum of the integers
******************************************************************************/
int Sum(int n)
{
	int i = n;	
	int sum = 0;
	
	for(; i > 0; i--)
	{
		sum += i;	
	}
	
	return sum;
}

* DESCRIPTION	: This function takes as input two parameters: the first
				  parameter is a positive integer L (length of a side), and the
				  second parameter a char type indicating the symbol s to
				  be used. The function draws an L x L square of s as well as
				  just the border of the square. 

				  Example:
				  Please enter a positive integer for L: 6

				  Please enter a symbol (single char): *

				  * * * * * *
				  * * * * * *

				  * * * * * *

#include <stdio.h> // printf, scanf


// function prototypes
void DrawSquare(int, char);
void DrawBorder(int, char);


// program entry
int main (void)
{
	
	int length = 0;
	char symbol = '*';
		
	printf("\n\nPlease enter a positive integer for L: ");
	scanf("%d", &length);
	
	printf("\nPlease enter a symbol (single char): ");
	scanf(" %c", &symbol);
	
	printf("\n\n");
	
	DrawSquare(length, symbol);
	DrawBorder(length, symbol);
		
	return 0;
}





/******************************************************************************
FUNCTION		: DrawSquare

DESCRIPTION		: This function will draw a filled length * length square with
				  the symbol supplied by the user.

INPUT			: Type			: int
				: Description	: the length of one of the sides of the square
				
OUTPUT			: Type 			: char
				: Description	: the character to print
******************************************************************************/
void DrawSquare(int length, char symbol)
{
	int i = 0;
	int j = 0;
	
	// vertical
	for(; i < length; i++)
	{
		// horizontal
		for(; j < length; j++)
			printf("%c ", symbol);
				
		// once all the character's have been printed from left to right
		// reset horizontal counter start a new line		
		j = 0;
		printf("\n");
	}
	
	printf("\n\n");
}





/******************************************************************************
FUNCTION		: DrawBorder

DESCRIPTION		: This function will draw a length * length border with
				  the symbol supplied by the user.

INPUT			: Type			: int
				: Description	: the length of one of the sides of the border
				
OUTPUT			: Type 			: char
				: Description	: the character to print
******************************************************************************/
void DrawBorder(int length, char symbol)
{
	int i = 0;
	int j = 0;
	int height = length;	
	
	// first and last run will fill an entire line
	for(; i < length; i++)
		printf("%c ", symbol);
		
	printf("\n");	
	height--;
	i = 0;
	
	// while we aren't at the bottom
	while(height > 1)
	{
		// print the first character on left, 
		// skip length - 2, then print the character again
		printf("%c ", symbol);
				
		// print the spaces in between each character		
		for(; i < length - 2; i++)
			printf("  ");
				
		// print the left side character		
		printf("%c\n", symbol);
		
		height--;
		i = 0;
	}	
	
	i = 0;
	
	// print the bottom border line
	for(; i < length; i++)
		printf("%c ", symbol);
		
	printf("\n\n");
}
